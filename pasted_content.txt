let lastPrices = {};

// دالة الحساب
function calculate(price, currency) {
    const gram24 = price / 31.1034768;
    lastPrices = {
        24: gram24,
        22: gram24 * 22 / 24,
        21: gram24 * 21 / 24,
        18: gram24 * 18 / 24
    };
    
    document.getElementById("ounce").innerText = price.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2}) + " " + currency;
    document.getElementById("g24").innerText = lastPrices[24].toFixed(2) + " $";
    document.getElementById("g22").innerText = lastPrices[22].toFixed(2) + " $";
    document.getElementById("g21").innerText = lastPrices[21].toFixed(2) + " $";
    document.getElementById("g18").innerText = lastPrices[18].toFixed(2) + " $";
}

// جلب السعر من Binance (غير محدود ومجاني)
async function loadGoldPrice() {
    try {
        const response = await fetch("https://api.binance.com/api/v3/ticker/price?symbol=PAXGUSDT");
        const data = await response.json();
        const goldPrice = parseFloat(data.price);
        
        if (goldPrice) {
            calculate(goldPrice, "USD");
            const tag = document.getElementById("statusTag");
            tag.innerText = "● تحديث مباشر من السوق العالمي";
            tag.style.color = "#27ae60";
            document.getElementById("error").innerText = "";
        }
    } catch (err) {
        document.getElementById("error").innerText = "❌ خطأ في الشبكة - استخدم اليدوي";
    }
}

function applyManual() {
    const v = parseFloat(document.getElementById("manualPrice").value);
    if (!v) return;
    calculate(v, "USD");
    const tag = document.getElementById("statusTag");
    tag.innerText = "⚠ السعر مُدخل يدويًا";
    tag.style.color = "#e67e22";
}

function convertAll() {
    const val = parseFloat(document.getElementById("convertValue").value);
    if (!val || !lastPrices[24]) return;

    const type = document.getElementById("convertType").value;
    const resDiv = document.getElementById("convertResult");
    resDiv.style.display = "block";
    let result = "";
    let values = [];

    for (let k in lastPrices) {
        let resVal = type === "moneyToGold" 
            ? (val / lastPrices[k]).toFixed(3) + " غرام" 
            : (val * lastPrices[k]).toFixed(2) + " $";

        result += `عيار ${k}: <b>${resVal}</b><br>`;
        values.push(type === "moneyToGold" ? val / lastPrices[k] : val * lastPrices[k]);
    }

    resDiv.innerHTML = result;
    drawChart(values);
    saveHistory(type === "moneyToGold" ? val + "$ إلى ذهب" : val + " غرام إلى $");
}

function drawChart(vals) {
    const c = document.getElementById("chart");
    const ctx = c.getContext("2d");
    ctx.clearRect(0,0,c.width,c.height);
    const max = Math.max(...vals);
    const barWidth = 40;
    vals.forEach((v,i)=>{
        const h = (v/max)*100;
        ctx.fillStyle="#d4af37";
        ctx.roundRect(50+i*70, 120-h, barWidth, h, 5);
        ctx.fill();
        ctx.fillStyle="#7f8c8d";
        ctx.font = "bold 11px Arial";
        ctx.fillText(["ع24","ع22","ع21","ع18"][i], 58+i*70, 140);
    });
}

// استخدام polyfill لـ roundRect في حال المتصفحات القديمة
if (!Path2D.prototype.roundRect) {
    CanvasRenderingContext2D.prototype.roundRect = function (x, y, w, h, r) {
        if (w < 2 * r) r = w / 2;
        if (h < 2 * r) r = h / 2;
        this.beginPath();
        this.moveTo(x + r, y);
        this.arcTo(x + w, y, x + w, y + h, r);
        this.arcTo(x + w, y + h, x, y + h, r);
        this.arcTo(x, y + h, x, y, r);
        this.arcTo(x, y, x + w, y, r);
        this.closePath();
        return this;
    };
}

function saveHistory(text) {
    let h = JSON.parse(localStorage.getItem("history")||"[]");
    const time = new Date().toLocaleTimeString('ar-EG', {hour: '2-digit', minute:'2-digit'});
    h.unshift(`[${time}] ${text}`);
    h = h.slice(0,3);
    localStorage.setItem("history",JSON.stringify(h));
    showHistory();
}

function showHistory() {
    const h = JSON.parse(localStorage.getItem("history")||"[]");
    document.getElementById("history").innerHTML = h.length > 0 ? "<b>🕒 السجل الأخير:</b><br>" + h.join("<hr style='margin:5px 0; opacity:0.05'>") : "";
}
